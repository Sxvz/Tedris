1   package sxvz.tedris.logic;
2   
3   import sxvz.tedris.domain.Palikka;
4   import sxvz.tedris.domain.Palikkakokoelma;
5   import sxvz.tedris.domain.Suunta;
6   import sxvz.tedris.engine.Pelilooppi;
7   
8   public class Vapaustarkastaja {
9   
10      private Pelilooppi peli;
11      private int pelialueenKorkeus;
12      private int pelialueenLeveys;
13  
14      public Vapaustarkastaja(Pelilooppi peli) {
15          this.peli = peli;
16          pelialueenKorkeus = peli.getPelialueenKorkeus();
17          pelialueenLeveys = peli.getPelialueenLeveys();
18      }
19  
20      public boolean voikoKokoelmaLiikkua(Palikkakokoelma k, Suunta s) {
21          for (Palikka p : k.getPalikat()) {   
22              if (!voikoLiikkua(k, p, s)) {
23                  return false;
24              }
25          }
26          return true;
27      }
28  
29      public boolean voikoLiikkua(Palikkakokoelma k, Palikka p, Suunta s) {
30          int x = p.getX();
31          int y = p.getY();
32  
33          if (s == Suunta.ALAS) {
34              return voikoLiikkua(k, x, y + 1);
35          } else if (s == Suunta.VASEN) {
36              return voikoLiikkua(k, x - 1, y);
37          } else if (s == Suunta.OIKEA) {
38              return voikoLiikkua(k, x + 1, y);
39          }
40          return false;
41      }
42  
43      public boolean voikoLiikkua(Palikkakokoelma liikkuvanKokoelma, int x, int y) {
44          
45          if (x <= -1 || x >= pelialueenLeveys || y <= -1 || y >= pelialueenKorkeus) {
46              return false;
47          }
48  
49          for (Palikkakokoelma palikkakokoelma : peli.getPalikat()) {
50              if (palikkakokoelma.equals(liikkuvanKokoelma)) {
51                  continue;
52              }
53              for (Palikka palikka : palikkakokoelma.getPalikat()) {
54                  if (palikka.getX() == x && palikka.getY() == y) {
55                      return false;
56                  }
57              }
58          }
59  
60          return true;
61      }
62      
63      public boolean voikoKokoelmaKaantya(Palikkakokoelma k, int kiertosuunta) {
64          if (k.getKaantymisInfo() == null) {
65              return false;
66          }
67          
68          Palikka p;
69          int haettavaKaantymisInfo;
70          for (int i = 1; i < 4; i++) {
71              p = k.getPalikat().get(i);
72              haettavaKaantymisInfo = k.getOrientaatio() + kiertosuunta;
73              
74              if (haettavaKaantymisInfo > 3) {
75                  haettavaKaantymisInfo = 0;
76              } else if (haettavaKaantymisInfo < 0) {
77                  haettavaKaantymisInfo = 3;
78              }
79              
80              if (!voikoLiikkua(k,
81                      k.getPalikat().get(0).getX() + k.getKaantymisInfo().get(haettavaKaantymisInfo).get(i - 1)[0],
82                      k.getPalikat().get(0).getY() + k.getKaantymisInfo().get(haettavaKaantymisInfo).get(i - 1)[1])) {
83                  return false;
84              }
85          }
86          
87          return true;
88      }
89  }
