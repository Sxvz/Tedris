2   package sxvz.tedris.logic;
3   
4   import java.util.ArrayList;
5   import sxvz.tedris.domain.Palikka;
6   import sxvz.tedris.domain.Palikkakokoelma;
7   import sxvz.tedris.engine.Pelilooppi;
8   
9   public class TaysienRivienTunnistaja {
10      private Pelilooppi peli;
11      
12      public TaysienRivienTunnistaja(Pelilooppi peli) {
13          this.peli = peli;
14      }
15      
16      public ArrayList<Integer> etsiTaydetRivit() {
17          ArrayList<Integer> taydetRivit = new ArrayList<>();
18          int[] palikatRivilla = new int[peli.getPelialueenKorkeus()];
19          
20          for (Palikkakokoelma kokoelma : peli.getPalikat()) {
21              for (Palikka p : kokoelma.getPalikat()) {
22                  palikatRivilla[p.getY()]++;
23              }
24          }
25          
26          for (int i = 0; i < palikatRivilla.length; i++) {
27              if (palikatRivilla[i] == peli.getPelialueenLeveys()) {
28                  taydetRivit.add(i);
29              }
30          }
31          
32          return taydetRivit;
33      }
34  }
